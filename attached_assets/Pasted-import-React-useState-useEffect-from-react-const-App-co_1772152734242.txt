import React, { useState, useEffect } from 'react';

const App = () => {
  const [activeStep, setActiveStep] = useState(0);

  const steps = [
    { id: 'idea', label: 'Идея', color: '#C6A848' },
    { id: 'synthesis', label: 'Синтез', color: '#8AE25D' },
    { id: 'animation', label: 'Оживление', color: '#48A2FB' },
    { id: 'launch', label: 'Запуск', color: '#31D8FA' }
  ];

  // Автоматическое проигрывание анимации
  useEffect(() => {
    const interval = setInterval(() => {
      setActiveStep((activeStep + 1) % steps.length);
    }, 3000); // смена шага каждые 3 секунды
    return () => clearInterval(interval);
  }, [activeStep, steps.length]);

  return (
    <div className="min-h-screen bg-[#f3f4f6] flex items-center justify-center p-4 font-sans text-gray-900">
      
      {/* Стили для SVG анимаций */}
      <style dangerouslySetInnerHTML={{__html: `
        /* 1. Идея (Лампочка) */
        .active-svg .bulb-glow {
          animation: pulse-glow 2s infinite alternate;
        }
        .active-svg .bulb-ray {
          animation: ray-flash 2s infinite;
          transform-origin: center;
        }
        @keyframes pulse-glow {
          0% { fill: transparent; }
          100% { fill: currentColor; opacity: 0.2; }
        }
        @keyframes ray-flash {
          0%, 100% { opacity: 0; transform: scale(0.8); }
          50% { opacity: 1; transform: scale(1.1); }
        }

        /* 2. Синтез (Узлы связи) */
        .active-svg .synth-line {
          stroke-dasharray: 10;
          animation: dash-flow 1.5s linear infinite;
        }
        .active-svg .synth-node {
          animation: node-pulse 1.5s infinite alternate;
        }
        @keyframes dash-flow {
          to { stroke-dashoffset: -20; }
        }
        @keyframes node-pulse {
          0% { r: 2.5; }
          100% { r: 3.5; }
        }

        /* 3. Оживление (Волшебная палочка / Искра) */
        .active-svg .wand {
          animation: wand-wave 2s infinite ease-in-out;
          transform-origin: bottom left;
        }
        .active-svg .sparkle {
          animation: sparkle-pop 1.5s infinite;
          transform-origin: center;
        }
        .active-svg .sparkle:nth-child(2) { animation-delay: 0.2s; }
        .active-svg .sparkle:nth-child(3) { animation-delay: 0.4s; }
        @keyframes wand-wave {
          0%, 100% { transform: rotate(0deg); }
          50% { transform: rotate(-15deg); }
        }
        @keyframes sparkle-pop {
          0%, 100% { opacity: 0; transform: scale(0) rotate(0deg); }
          50% { opacity: 1; transform: scale(1) rotate(45deg); }
        }

        /* 4. Запуск (Ракета) */
        .active-svg .rocket-body {
          animation: rocket-shake 0.5s infinite;
        }
        .active-svg .rocket-flame {
          animation: flame-flicker 0.1s infinite alternate;
          transform-origin: top;
        }
        @keyframes rocket-shake {
          0%, 100% { transform: translate(0, 0); }
          25% { transform: translate(0.5px, -0.5px); }
          50% { transform: translate(-0.5px, -1px); }
          75% { transform: translate(0, -0.5px); }
        }
        @keyframes flame-flicker {
          0% { transform: scaleY(0.8); opacity: 0.8; }
          100% { transform: scaleY(1.2); opacity: 1; }
        }
      `}} />

      {/* Карточка */}
      <div className="bg-white rounded-[2rem] shadow-[0_10px_40px_-15px_rgba(0,0,0,0.05)] max-w-4xl w-full p-12 md:p-16">

        {/* Дорожная карта */}
        <div className="relative pt-8 pb-12">
          
          {/* Линии фона и прогресса */}
          <div className="absolute top-[42px] left-0 right-0 h-1 bg-gray-100 rounded-full z-0"></div>
          <div 
            className="absolute top-[42px] left-0 right-0 h-1 rounded-full z-0 transition-all duration-700 ease-in-out"
            style={{ 
              backgroundImage: 'linear-gradient(90deg, #C6A848 0%, #8AE25D 33%, #48A2FB 66%, #31D8FA 100%)',
              clipPath: `inset(0 ${100 - (activeStep / (steps.length - 1)) * 100}% 0 0)`
            }}
          ></div>

          {/* Этапы */}
          <div className="relative z-10 flex justify-between">
            {steps.map((step, index) => {
              const isActive = index === activeStep;
              const isPassed = index <= activeStep;
              
              return (
                <div 
                  key={step.id}
                  className="flex flex-col items-center cursor-pointer group"
                  onClick={() => setActiveStep(index)}
                >
                  {/* Иконка-контейнер */}
                  <div className={`
                    w-12 h-12 rounded-full flex items-center justify-center bg-white transition-all duration-500
                    border-[3px] 
                    ${isActive ? 'scale-110' : ''}
                    ${!isActive && !isPassed ? 'border-gray-200 text-gray-300 group-hover:border-gray-300 group-hover:text-gray-400' : ''}
                  `}
                  style={
                    isActive || isPassed
                      ? {
                          borderColor: step.color,
                          color: step.color,
                          boxShadow: isActive ? `0 0 15px ${step.color}66` : 'none'
                        }
                      : {}
                  }>
                    
                    {/* SVG 1: Идея */}
                    {index === 0 && (
                      <svg viewBox="0 0 24 24" fill="none" stroke="currentColor" strokeWidth="2" strokeLinecap="round" strokeLinejoin="round" className={`w-6 h-6 ${isActive ? 'active-svg' : ''}`}>
                        <path className="bulb-glow" d="M12 2a5 5 0 0 0-5 5c0 1.5.8 2.8 1.8 3.5C9.6 11.1 10 12 10 13v3h4v-3c0-1 .4-1.9 1.2-2.5C16.2 9.8 17 8.5 17 7a5 5 0 0 0-5-5Z" />
                        <path d="M12 2a5 5 0 0 0-5 5c0 1.5.8 2.8 1.8 3.5C9.6 11.1 10 12 10 13v3h4v-3c0-1 .4-1.9 1.2-2.5C16.2 9.8 17 8.5 17 7a5 5 0 0 0-5-5Z" />
                        <path d="M9 18h6" />
                        <path d="M10 21h4" />
                        {/* Лучи */}
                        <path className="bulb-ray" d="M12 0v1" style={{animationDelay: '0s'}} />
                        <path className="bulb-ray" d="M5 4l1 1" style={{animationDelay: '0.2s'}} />
                        <path className="bulb-ray" d="M19 4l-1 1" style={{animationDelay: '0.4s'}} />
                      </svg>
                    )}

                    {/* SVG 2: Синтез */}
                    {index === 1 && (
                      <svg viewBox="0 0 24 24" fill="none" stroke="currentColor" strokeWidth="2" strokeLinecap="round" strokeLinejoin="round" className={`w-6 h-6 ${isActive ? 'active-svg' : ''}`}>
                        <path className="synth-line" d="M17 7L7 12l10 5" />
                        <circle className="synth-node" cx="17" cy="7" r="3" fill={isActive ? 'currentColor' : 'white'} />
                        <circle className="synth-node" cx="7" cy="12" r="3" fill={isActive ? 'currentColor' : 'white'} style={{animationDelay: '0.5s'}} />
                        <circle className="synth-node" cx="17" cy="17" r="3" fill={isActive ? 'currentColor' : 'white'} style={{animationDelay: '1s'}} />
                      </svg>
                    )}

                    {/* SVG 3: Оживление */}
                    {index === 2 && (
                      <svg viewBox="0 0 24 24" fill="none" stroke="currentColor" strokeWidth="2" strokeLinecap="round" strokeLinejoin="round" className={`w-6 h-6 ${isActive ? 'active-svg' : ''}`}>
                        <path className="wand" d="M15 4V2M15 16v-2M8 9h2M20 9h2M17.8 11.8L19 13M10.2 6.2L9 5M17.8 6.2L19 5" />
                        <path className="wand" d="m3 21 9-9" />
                        <path className="wand" d="M12.2 9.8l2 2" />
                        {/* Искры */}
                        <path className="sparkle" d="M16 11l-1-1 1-1 1 1-1 1z" fill="currentColor"/>
                        <path className="sparkle" d="M12 5l-1-1 1-1 1 1-1 1z" fill="currentColor"/>
                        <path className="sparkle" d="M21 9l-1-1 1-1 1 1-1 1z" fill="currentColor"/>
                      </svg>
                    )}

                    {/* SVG 4: Запуск */}
                    {index === 3 && (
                      <svg viewBox="0 0 24 24" fill="none" stroke="currentColor" strokeWidth="2" strokeLinecap="round" strokeLinejoin="round" className={`w-6 h-6 ${isActive ? 'active-svg' : ''}`}>
                        <g className="rocket-body">
                          <path d="M4.5 16.5c-1.5 1.26-2 5-2 5s3.74-.5 5-2c.71-.84.7-2.13-.09-2.91a2.18 2.18 0 0 0-2.91-.09z"/>
                          <path d="m12 15-3-3a22 22 0 0 1 2-3.95A12.88 12.88 0 0 1 22 2c0 2.72-.78 7.5-6 11a22.35 22.35 0 0 1-4 2z"/>
                          <path d="M9 12H4s.55-3.03 2-4c1.62-1.08 5 0 5 0"/>
                          <path d="M12 15v5s3.03-.55 4-2c1.08-1.62 0-5 0-5"/>
                        </g>
                        <path className="rocket-flame" d="M8 16l-2 4 1-1 1 2 2-3" fill={isActive ? 'currentColor' : 'none'} stroke="none"/>
                      </svg>
                    )}
                  </div>
                  
                  {/* Текст под иконкой */}
                  <span className={`
                    mt-4 font-semibold text-sm transition-colors duration-300
                    ${isPassed ? 'text-gray-900' : 'text-gray-400'}
                  `}>
                    {step.label}
                  </span>
                </div>
              );
            })}
          </div>
        </div>
      </div>
    </div>
  );
};

export default App;
